75void
76UdpEchoServer::StartApplication()
77{
78 NS_LOG_FUNCTION(this);
79
80 if (!m_socket)
81 {
82 TypeId tid = TypeId::LookupByName("ns3::UdpSocketFactory");
83 m_socket = Socket::CreateSocket(GetNode(), tid);
84 InetSocketAddress local = InetSocketAddress(Ipv4Address::GetAny(), m_port);
85 if (m_socket->Bind(local) == -1)
86 {
87 NS_FATAL_ERROR("Failed to bind socket");
88 }
89 if (addressUtils::IsMulticast(m_local))
90 {
91 Ptr<UdpSocket> udpSocket = DynamicCast<UdpSocket>(m_socket);
92 if (udpSocket)
93 {
94 // equivalent to setsockopt (MCAST_JOIN_GROUP)
95 udpSocket->MulticastJoinGroup(0, m_local);
96 }
97 else
98 {
99 NS_FATAL_ERROR("Error: Failed to join multicast group");
100 }
101 }
102 }
103
104 if (!m_socket6)
105 {
106 TypeId tid = TypeId::LookupByName("ns3::UdpSocketFactory");
107 m_socket6 = Socket::CreateSocket(GetNode(), tid);
108 Inet6SocketAddress local6 = Inet6SocketAddress(Ipv6Address::GetAny(), m_port);
109 if (m_socket6->Bind(local6) == -1)
110 {
111 NS_FATAL_ERROR("Failed to bind socket");
112 }
113 if (addressUtils::IsMulticast(local6))
114 {
115 Ptr<UdpSocket> udpSocket = DynamicCast<UdpSocket>(m_socket6);
116 if (udpSocket)
117 {
118 // equivalent to setsockopt (MCAST_JOIN_GROUP)
119 udpSocket->MulticastJoinGroup(0, local6);
120 }
121 else
122 {
123 NS_FATAL_ERROR("Error: Failed to join multicast group");
124 }
125 }
126 }
127
128 m_socket->SetIpTos(m_tos); // Affects only IPv4 sockets.
129 m_socket->SetRecvCallback(MakeCallback(&UdpEchoServer::HandleRead, this));
130 m_socket6->SetRecvCallback(MakeCallback(&UdpEchoServer::HandleRead, this));
131}
